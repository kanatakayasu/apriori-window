\section{問題定義}\label{sec:problem}
% ==============================================================

\subsection{記法}

$\mathcal{I} = \{i_1, i_2, \ldots, i_m\}$ を有限のアイテム集合とする。
$D = [d_1, d_2, \ldots, d_N]$ を時間インデックス $t \in \{1, \ldots, N\}$ で順序付けられたトランザクション列とする。
各トランザクション $d_t \subseteq \mathcal{I}$ はアイテムの集合である。
アイテムセット（パターン）$P \subseteq \mathcal{I}$ がトランザクション $d_t$ に\emph{出現する}とは、$P \subseteq d_t$ が成り立つことをいう。

$k$ 個の補助次元 $x_1, x_2, \ldots, x_k$（例：空間座標、店舗識別子、属性インデックス）を考え、各次元は有限順序付き定義域 $\mathcal{X}_j$（$j = 1, \ldots, k$）の値をとる。
\emph{次元空間} $\mathcal{X} = \mathcal{X}_1 \times \mathcal{X}_2 \times \cdots \times \mathcal{X}_k$ を定義する。
各トランザクション $d_t$ は位置 $\mathbf{x}(t) = (x_1(t), \ldots, x_k(t)) \in \mathcal{X}$ に関連付けられる。

\subsection{サポート曲面}

\begin{definition}[多次元サポート曲面]\label{def:support_surface}
パターン $P$、時間窓サイズ $W_0 \in \mathbb{Z}_{>0}$、および空間窓サイズ $W_j \in \mathbb{Z}_{>0}$（$j = 1, \ldots, k$）が与えられたとき、$P$ の\emph{サポート曲面}は関数 $S_P : \mathbb{Z}^{k+1} \to \mathbb{Z}_{\geq 0}$ であり、以下で定義される：
\begin{equation}\label{eq:support_surface}
S_P(t, \mathbf{v}) = \bigl|\{d_r \in D \mid t \leq r < t + W_0,\; P \subseteq d_r,\; \mathbf{x}(r) \in \mathrm{Box}(\mathbf{v}, \mathbf{W})\}\bigr|,
\end{equation}
ここで $\mathbf{v} = (v_1, \ldots, v_k) \in \mathcal{X}$、$\mathbf{W} = (W_1, \ldots, W_k)$、かつ
\begin{equation}
\mathrm{Box}(\mathbf{v}, \mathbf{W}) = \{\mathbf{u} \in \mathcal{X} \mid v_j \leq u_j < v_j + W_j,\; j = 1, \ldots, k\}
\end{equation}
である。
$S_P$ の定義域は、窓がデータ境界内に収まるすべての $(t, \mathbf{v})$ の集合である。
$k = 0$（補助次元なし）のとき、$S_P(t)$ は Apriori-Window の標準的な1次元サポート系列に帰着する。
\end{definition}

\begin{remark}
サポート曲面は $(k{+}1)$ 次元テンソル $\mathcal{S}_P \in \mathbb{Z}_{\geq 0}^{T_0 \times T_1 \times \cdots \times T_k}$ として見ることができる。ここで $T_j = |\mathcal{X}_j| - W_j + 1$ は次元 $j$ に沿った有効な窓位置の数（$T_0 = N - W_0 + 1$）である。
\end{remark}

\subsection{密集領域}

\begin{definition}[密集領域]\label{def:dense_region}
サポート曲面 $S_P$ と密度閾値 $\theta \in \mathbb{Z}_{>0}$ が与えられたとき、閾値 $\theta$ における\emph{上位レベル集合}は
\begin{equation}\label{eq:superlevel}
\mathcal{L}_\theta(S_P) = \{(t, \mathbf{v}) \mid S_P(t, \mathbf{v}) \geq \theta\}
\end{equation}
である。
閾値 $\theta$ における $P$ の\emph{密集領域}は、グリッド隣接性（2つのグリッド点がちょうど1つの座標で1だけ異なる場合に隣接する）のもとでの $\mathcal{L}_\theta(S_P)$ の極大連結成分である。
\end{definition}

\begin{definition}[密集領域集合]\label{def:dense_region_set}
$\mathcal{R}_\theta(P) = \{R_1, R_2, \ldots, R_q\}$ を閾値 $\theta$ におけるパターン $P$ のすべての密集領域の集合とする。
各 $R_i \subseteq \mathbb{Z}^{k+1}$ はグリッド点の連結集合である。
\end{definition}

\begin{remark}[1次元密集区間との関係]
$k = 0$ のとき、密集領域 $R_i$ は連続する時間インデックスの集合 $\{s, s{+}1, \ldots, e\}$ であり、元の Apriori-Window の定式化における密集区間 $[s, e]$ に対応する。
したがって、密集領域は密集区間を厳密に多次元に一般化するものである。
\end{remark}

\subsection{密集領域包含定理}

\begin{theorem}[密集領域包含]\label{thm:containment}
$P \subseteq \mathcal{I}$ を $|P| \geq 2$ のパターンとし、任意の $i \in P$ に対して $Q = P \setminus \{i\}$ とする。
このとき、すべての $(t, \mathbf{v})$ に対して：
\begin{equation}
S_P(t, \mathbf{v}) \leq S_Q(t, \mathbf{v}).
\end{equation}
したがって、
\begin{equation}\label{eq:containment}
\mathcal{L}_\theta(S_P) \subseteq \mathcal{L}_\theta(S_Q),
\end{equation}
かつ $P$ のすべての密集領域は $Q$ のいずれかの密集領域に含まれる：
\begin{equation}
\forall R \in \mathcal{R}_\theta(P),\;\exists R' \in \mathcal{R}_\theta(Q) \text{ such that } R \subseteq R'.
\end{equation}
\end{theorem}

\begin{proof}
$Q \subset P$ であるから、$P \subseteq d_r$ を満たす任意のトランザクション $d_r$ は $Q \subseteq d_r$ も満たす。
したがって、$S_P(t, \mathbf{v})$ で計数されるすべてのトランザクションは $S_Q(t, \mathbf{v})$ でも計数され、$S_P(t, \mathbf{v}) \leq S_Q(t, \mathbf{v})$ が得られる。

上位レベル集合の包含は直接的に従う：$S_P(t, \mathbf{v}) \geq \theta$ ならば $S_Q(t, \mathbf{v}) \geq \theta$ である。

密集領域の包含について、$R \in \mathcal{R}_\theta(P)$ とする。
$R$ のすべての点は式~\eqref{eq:containment}により $\mathcal{L}_\theta(S_Q)$ に属する。
$R$ はグリッド隣接性のもとで連結であるから、$R$ のすべての点は $\mathcal{L}_\theta(S_Q)$ の単一の連結成分に含まれる。すなわち、ある $R' \in \mathcal{R}_\theta(Q)$ の中に含まれる。
\end{proof}

\begin{corollary}[多次元Apriori枝刈り]\label{cor:pruning}
$P$ の任意の $(|P|{-}1)$-部分集合 $Q \subset P$ に対して $\mathcal{R}_\theta(Q) = \emptyset$ であれば、$\mathcal{R}_\theta(P) = \emptyset$ である。
\end{corollary}

\begin{corollary}[候補領域の制限]\label{cor:candidate_restriction}
$k$-アイテムセット $P$ を評価する際、サポート曲面 $S_P$ はすべての $(|P|{-}1)$-部分集合の上位レベル集合の共通部分内でのみ計算すればよい：
\begin{equation}
\mathcal{L}_\theta(S_P) \subseteq \bigcap_{Q \subset P,\, |Q| = |P|-1} \mathcal{L}_\theta(S_Q).
\end{equation}
\end{corollary}

\subsection{次元分解可能性}

\begin{definition}[周辺サポート]\label{def:marginal}
次元 $j$ に沿った位置 $v_j$ における $P$ の\emph{周辺サポート}は
\begin{equation}
S_P^{(j)}(v_j) = \sum_{\substack{(t, \mathbf{v}) \\ \text{proj}_j(\mathbf{v}) = v_j}} S_P(t, \mathbf{v})
\end{equation}
であり、ここで $\text{proj}_j$ は第 $j$ 座標を抽出する。
\end{definition}

\begin{definition}[次元分解可能性]\label{def:decomposable}
サポート曲面 $S_P$ が\emph{次元分解可能}であるとは、関数 $f_0 : \mathbb{Z} \to \mathbb{R}_{\geq 0}$ および $f_j : \mathcal{X}_j \to \mathbb{R}_{\geq 0}$（$j = 1, \ldots, k$）が存在して
\begin{equation}\label{eq:decomposable}
S_P(t, \mathbf{v}) = f_0(t) \cdot \prod_{j=1}^{k} f_j(v_j)
\end{equation}
が成り立つことをいう。
\end{definition}

\begin{theorem}[分解定理]\label{thm:decomposition}
$S_P$ が $S_P(t, \mathbf{v}) = f_0(t) \prod_{j=1}^{k} f_j(v_j)$ と次元分解可能であり、$\theta$ が $\theta = \theta_0 \cdot \prod_{j=1}^{k} \theta_j$ と分解可能であるとする。このとき
\begin{equation}
\mathcal{L}_\theta(S_P) = \{t : f_0(t) \geq \theta_0\} \times \prod_{j=1}^{k} \{v_j : f_j(v_j) \geq \theta_j\}.
\end{equation}
この場合、密集領域検出は $k{+}1$ 個の独立な1次元問題に分解され、それぞれ $O(T_j)$ の時間で解ける。
\end{theorem}

\begin{proof}
$S_P(t, \mathbf{v}) \geq \theta$ は $f_0(t) \prod_j f_j(v_j) \geq \theta_0 \prod_j \theta_j$ と同値である。
すべての因子が非負であるから、これは $\theta$ の乗法的分解により $f_0(t) \geq \theta_0$ かつすべての $j$ に対して $f_j(v_j) \geq \theta_j$ が成り立つことと同値である。
したがって、上位レベル集合は1次元の上位レベル集合の直積となる。
\end{proof}

\begin{remark}[実用上の意義]
厳密な分解可能性が実際に成り立つことは稀である。しかし、近似的な分解可能性（ランク1テンソル近似の残差で測定）は依然として有用な高速化をもたらしうる。
\ref{sec:method}節では、まず近似的な分解可能性を確認し、近似誤差がユーザー指定の許容値 $\epsilon$ を超える場合に完全な Sweep Surface 手法にフォールバックするハイブリッドアルゴリズムを提案する。
\end{remark}

\subsection{計算量解析}

\begin{theorem}[密集領域検出の計算量]\label{thm:complexity}
$G = \prod_{j=0}^{k} T_j$ をグリッド点の総数とする。

\begin{enumerate}
\item \textbf{素朴なアルゴリズム}：各グリッド点に対してすべてのトランザクションを走査してサポート曲面を計算するには $O(G \cdot N)$ の時間を要する。

\item \textbf{接頭辞和による構築}：サポート曲面は $(k{+}1)$ 次元の接頭辞和に対する包除原理により $O(G \cdot 2^{k+1} + N)$ の時間と $O(G)$ の空間で計算できる。

\item \textbf{密集領域の抽出}：サポート曲面が与えられた場合、Union-Findによる連結成分ラベリングは $O(G \cdot \alpha(G))$ の時間を要する。ここで $\alpha$ は逆Ackermann関数である。

\item \textbf{分解可能な場合}：次元分解可能な場合、全体の計算量は $O(\sum_{j=0}^{k} T_j + N)$ である。
\end{enumerate}
\end{theorem}

\subsection{問題の定式化}

\begin{problem}[多次元密集アイテムセットマイニング]\label{prob:main}
\textbf{入力}：関連する位置 $\mathbf{x}(t)$ を持つトランザクション列 $D$、窓サイズ $(W_0, W_1, \ldots, W_k)$、密度閾値 $\theta$、最大パターン長 $L_{\max}$。

\textbf{出力}：$|P| \leq L_{\max}$ かつ $\mathcal{R}_\theta(P) \neq \emptyset$ であるすべてのパターン $P$ とその密集領域集合 $\mathcal{R}_\theta(P)$。
\end{problem}
