\section{問題定義}\label{sec:problem}
% ==============================================================

\subsection{記法}

$\mathcal{I} = \{i_1, i_2, \ldots, i_m\}$ を有限のアイテム全集合，
$D = [d_1, d_2, \ldots, d_N]$ を時刻順のトランザクション列とする．
各トランザクション $d_t$（$t = 1, \ldots, N$）は $\mathcal{I}$ の部分集合である．
アイテムセット（パターン）$P \subseteq \mathcal{I}$ が
トランザクション $d_t$ に\emph{出現}するとは $P \subseteq d_t$ が成立することをいう．

\begin{definition}[サポート時系列]\label{def:support_series}
パターン $P$，ウィンドウサイズ $W \in \mathbb{Z}_{>0}$，
トランザクション列 $D$（長さ $N$）に対して，
$P$ の\emph{サポート時系列} $s_P = [s_P(0), s_P(1), \ldots, s_P(N-W)]$ を
\begin{equation}
s_P(t) = \bigl|\{d_j \in D \mid t \leq j < t + W,\; P \subseteq d_j\}\bigr|
\label{eq:support_series}
\end{equation}
と定義する．これは位置 $t$ を始点とするウィンドウ $[t, t+W)$ 内で
$P$ を含むトランザクション数である．
\end{definition}

\begin{definition}[外部イベント]\label{def:event}
\emph{外部イベント} $e = (\textit{id}, \textit{name}, t_s, t_e)$ は，
一意な識別子，名称，およびイベントが有効な期間を表すタイムスタンプ区間 $[t_s, t_e]$ から成る．
既知の外部イベント集合を $\mathcal{E} = \{e_1, e_2, \ldots, e_K\}$ とする．
\end{definition}

\begin{definition}[変化点（閾値交差）]\label{def:change_point}
本稿では，サポート時系列 $s_P$ に対する\emph{変化点}として，
一般のレベルシフト点すべてではなく，
閾値 $\theta$ に関する密状態／非密状態の切替点のみを扱う．
位置 $\tau \in \{0, 1, \ldots, N-W\}$ が変化点であるとは，
次のいずれかが成り立つことをいう:
\begin{align}
s_P(\tau) \geq \theta \land (\tau = 0 \lor s_P(\tau-1) < \theta), \label{eq:up_crossing}\\
s_P(\tau) < \theta \land \tau > 0 \land s_P(\tau-1) \geq \theta. \label{eq:down_crossing}
\end{align}
式~(\ref{eq:up_crossing}) を満たすとき方向を「上昇」，
式~(\ref{eq:down_crossing}) を満たすとき方向を「下降」とする．

さらに，$T = N-W+1$ とし，
\begin{align*}
\mathcal{L}^{-}_{\tau} &= \{\max(0, \tau-h), \ldots, \tau-1\},\\
\mathcal{L}^{+}_{\tau} &= \{\tau, \ldots, \min(T-1, \tau+h-1)\}
\end{align*}
を交差前後の参照窓とする．
変化量はこれらの局所平均差
\begin{align*}
\bar{s}^{-}(\tau) &=
\frac{1}{\max(1, |\mathcal{L}^{-}_{\tau}|)}
\sum_{j \in \mathcal{L}^{-}_{\tau}} s_P(j), \\
\bar{s}^{+}(\tau) &=
\frac{1}{\max(1, |\mathcal{L}^{+}_{\tau}|)}
\sum_{j \in \mathcal{L}^{+}_{\tau}} s_P(j)
\end{align*}
に基づき
\begin{equation}
\text{magnitude}(\tau) = |\bar{s}^{+}(\tau) - \bar{s}^{-}(\tau)|
\end{equation}
と定義する．
したがって，密集区間がウィンドウ左端の連続区間 $[s, e]$ で与えられるとき，
対応する変化点は開始時刻 $s$ と，存在すれば終了直後の $e+1$ である．
\end{definition}

\begin{definition}[密集区間]\label{def:dense_interval}
パターン $P$ とサポート閾値 $\theta$ に対して，
\emph{密集区間} $[s, e] \subseteq \{0, \ldots, T-1\}$ とは，
$s_P(t) \geq \theta$ が $t \in [s, e]$ の全ての $t$ で成立し，
かつ $s < 0$ または $s_P(s-1) < \theta$，
$e = T-1$ または $s_P(e+1) < \theta$ を満たす極大連続区間である．
$P$ の密集区間全体からなる集合を $\mathcal{I}_P$ とする．
$\mathcal{I}_P$ は Step~0（Apriori-window~\cite{dsaa2025}，\ref{sec:step0}節参照）の出力として与えられる．
\end{definition}

\subsection{イベント帰属問題}

\begin{definition}[イベント帰属問題]\label{def:attribution}
共起パターン集合 $\mathcal{F} = \{P \subseteq \mathcal{I} : |P| \geq 2,\;
\text{サポート} \geq \theta\}$，各パターン $P$ の密集区間集合 $\{\mathcal{I}_P\}_{P \in \mathcal{F}}$，
外部イベント集合 $\mathcal{E}$，有意水準 $\alpha$ が与えられたとき，
\emph{イベント帰属問題}とは，
パターン $P$ の密集区間 $I \in \mathcal{I}_P$ の変化点が
イベント $e$ の時間窓と統計的に有意に関連する
全ての三つ組 $(P, I, e) \in \{(P, I, e) : P \in \mathcal{F},\; I \in \mathcal{I}_P,\; e \in \mathcal{E}\}$ を，
偽発見率（FDR）を水準 $\alpha$ 以下に制御しながら同定することである．
単一アイテム（$|P| = 1$）の頻度変動は通常の時系列分析で捕捉可能であり，
共起構造の発見を目的とする本問題の対象外とする．
\end{definition}

\begin{remark}[帰属の意味と限界]\label{rem:attribution_scope}
本稿における「帰属」は\textbf{統計的時間的関連}（statistical temporal association）の
有意性を指す．具体的には，帰無仮説
$H_0$:「密集区間 $I$ の変化点の位置はイベント $e$ の時刻と時系列上で無関係である」
を棄却できることを
「パターン $P$ の密集区間 $I$ がイベント $e$ に帰属される」と表現する．
この意味での「帰属」はイベント・スタディ法~\cite{mackinlay1997event}における
「異常リターンの帰属」と同様であり，
以下の2点を明示的に保証\textbf{しない}．

\begin{itemize}
\item \textbf{因果効果の推定}: Rubin の潜在結果フレームワーク~\cite{rubin1974estimating}
  が要求する反事実的問い（イベントがなければどうなっていたか）には答えない．
  Pearl の do-calculus~\cite{pearl2009causality} における介入分布 $P(Y \mid \mathrm{do}(X))$ ではなく
  観測分布 $P(Y \mid X)$ を評価する手法であり，
  交絡要因（観測期間中に同期する別の外生変動）が存在する場合，
  有意な帰属が必ずしも真の原因を指すとは限らない．

\item \textbf{複数イベント間の因果弁別}: 複数イベントの時間窓が重複する場合，
  本手法は近接度スコアが高いイベントへ帰属するが，
  どのイベントが実際に変動を引き起こしたかを弁別する能力は限定的である．
  この限界は CONFOUND・OVERLAP条件での精度低下として実験的に確認される
  （\ref{sec:exp_ex1}節）．
\end{itemize}

本手法の出力は帰属候補の\textbf{スクリーニング}として位置づけられ，
確定的な因果分析にはさらなる実験的介入または自然実験の設計が必要である．
\end{remark}

この問題の主要な課題は以下の3点である:

\begin{enumerate}
\item \textbf{規模}: $\sum_{P \in \mathcal{F}} |\mathcal{I}_P| \times |\mathcal{E}|$ 個の
$(P, I, e)$ 仮説を同時に検定する必要があり，厳格な多重検定補正が必要となる．
パターン数 $|\mathcal{F}|$ が数千，各パターンが複数の密集区間を持ちうる設定では，
この仮説空間は非常に大きくなる．

\item \textbf{副次パターン}: イベントがアイテム $i$ の頻度を増加させると，
$i$ を含む\emph{全て}の共起パターンのサポートが相関して変動し，偽陽性が大量に発生する．

\item \textbf{時間的混同}: 無関係なイベントの時間窓が重複する場合，帰属が曖昧になる．
\end{enumerate}
