\begin{algorithm}[!ht]
\footnotesize
\caption{イベント帰属パイプライン}
\label{alg:pipeline}
\begin{algorithmic}[1]
\Require 共起パターン $\mathcal{F}\;(|P| \geq 2)$, 密集区間 $\{\mathcal{I}_P\}$, イベント $\mathcal{E}$, 設定 $(\theta, \sigma, B, \alpha, \delta)$
\Ensure 有意帰属集合 $\mathcal{R}$
\Statex \textbf{// Step 1--2: 変化点抽出 + フィルタリング}
\State $\text{Raw} \gets []$
\For{$P \in \mathcal{F}$ with $|P| \geq 2$}
    \If{$\mathcal{I}_P = \emptyset$} \textbf{continue} \EndIf
    \State $\mathcal{C}_P \gets \{s_i, e_i{+}1 \mid [s_i, e_i] \in \mathcal{I}_P\}$ \Comment{端点から変化点}
    \State 各 $\tau \in \mathcal{C}_P$ の変化量を計算 \Comment{式~(\ref{eq:magnitude})}
    \If{$\mathcal{C}_P = \emptyset$} \textbf{continue} \EndIf
    \Statex \textbf{// Step 3--4: スコアリング + 置換検定}
    \For{$e \in \mathcal{E}$}
        \State $A_{\text{obs}} \gets \sum_{\tau} A(P, \tau, e)$; $c \gets 0$ \Comment{式~(\ref{eq:attribution_score})}
        \For{$b = 1$ to $B$}
            \State $\mathcal{E}_b \gets \Call{CircShift}{\mathcal{E}, \delta_b}$; $A_b \gets \sum_{\tau} A(P, \tau, e_b)$
            \State \textbf{if} $A_b \geq A_{\text{obs}}$ \textbf{then} $c \gets c + 1$
        \EndFor
        \State $p_{\text{raw}} \gets (1+c)/(B+1)$ \Comment{式~(\ref{eq:pvalue})}
        \State $\text{Raw}.\text{append}(P, e, A_{\text{obs}}, p_{\text{raw}})$
    \EndFor
\EndFor
\Statex \textbf{// Step 4 (続): グローバル BH 補正}
\State $\text{Raw}$ を $p_{\text{raw}}$ 昇順にソート; $M \gets |\text{Raw}|$
\For{$i = M$ downto $1$}
    $\tilde{p}_{(i)} \gets \min(\tilde{p}_{(i+1)}, \min(1, p_{(i)} M / i))$ \Comment{式~(\ref{eq:bh})}
\EndFor
\State $\mathcal{R} \gets \{(P, e) \mid \tilde{p}_{(i)} < \alpha\}$
\Statex \textbf{// Step 5a: 同一長の重複排除（過半数共有）}
\For{$e$: 帰属パターンを長さ $l$ ごとにグループ化}
    \State $|P_i \cap P_j| \geq \lceil l/2 \rceil$ で辺 $\to$ Union-Find; 各成分のスコア最大を残す
\EndFor
\Statex \textbf{// Step 5b: 異長間の部分集合排除}
\For{$e$: $P_i \subseteq P_j$ で辺 $\to$ Union-Find; 各成分のスコア最大を残す}
\EndFor
\State \Return $\mathcal{R}$
\end{algorithmic}
\end{algorithm}
