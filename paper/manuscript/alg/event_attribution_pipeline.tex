\begin{algorithm}[!ht]
\caption{イベント帰属パイプライン}
\label{alg:pipeline}
\begin{algorithmic}[1]
\Require 共起パターン $\mathcal{F}\;(|P| \geq 2)$, 密集区間 $\{\mathcal{I}_P\}$, イベント $\mathcal{E}$, 設定 $(\theta, \sigma, B, \alpha, \delta)$
\Ensure 有意帰属集合 $\mathcal{R}$
\Statex \textbf{// Step 1--2: 変化点抽出}
\State $\text{Raw} \gets []$
\For{$P \in \mathcal{F}$ with $\mathcal{I}_P \neq \emptyset$}
    \State $\mathcal{C}_P \gets \{s_i, e_i{+}1 \mid [s_i,e_i] \in \mathcal{I}_P\}$
    \State 各 $\tau \in \mathcal{C}_P$ の変化量を計算
    \Statex \textbf{// Step 3--4: スコアリング + 置換検定}
    \For{$e \in \mathcal{E}$}
        \State $A_{\text{obs}} \gets \sum_{\tau} A(P,\tau,e)$; $c \gets 0$
        \For{$b = 1$ to $B$}
            \State $A_b \gets \sum_{\tau} A(P,\tau,\Call{CircShift}{e,\delta_b})$
            \If{$A_b \geq A_{\text{obs}}$} $c \gets c+1$ \EndIf
        \EndFor
        \State $\text{Raw}.\text{append}(P,e,A_{\text{obs}},(1{+}c)/(B{+}1))$
    \EndFor
\EndFor
\Statex \textbf{// Step 4 (続): グローバル BH 補正}
\State $\text{Raw}$ を $p$ 昇順にソート; $M \gets |\text{Raw}|$
\For{$i = M$ downto $1$}
    $\tilde{p}_{(i)} \gets \min(\tilde{p}_{(i+1)}, p_{(i)} M/i)$
\EndFor
\State $\mathcal{R} \gets \{(P,e) \mid \tilde{p}_{(i)} < \alpha\}$
\Statex \textbf{// Step 5: 重複排除}（\ref{sec:dedup}節参照）
\State $\mathcal{R} \gets \Call{Dedup}{\mathcal{R}}$
\State \Return $\mathcal{R}$
\end{algorithmic}
\end{algorithm}
