\section{提案手法}
\label{sec:method}

\subsection{理論的基礎}

まず、効率的な枝刈りを可能にする性質を確立する。

\begin{property}[TWU上界]
\label{prop:twu-bound}
任意のアイテムセット$X$およびウィンドウ位置$l$に対して：
\begin{equation}
    \text{WU}(X, l) \leq \text{WTWU}(X, l)
\end{equation}
\end{property}

\begin{proof}
$l \leq t \leq l + W$かつ$X \subseteq I(T_t)$を満たす各トランザクション$T_t$に対して：
\[
u(X, T_t) = \sum_{i \in X} u(i, T_t) \leq \sum_{i \in I(T_t)} u(i, T_t) = \text{TU}(T_t)
\]
該当するすべてのトランザクションについて総和をとると$\text{WU}(X, l) \leq \text{WTWU}(X, l)$を得る。
\end{proof}

\begin{lemma}[WTWUの反単調性]
\label{lem:twu-antimonotone}
任意のアイテムセット$X \subseteq Y$およびウィンドウ位置$l$に対して：
\begin{equation}
    \text{WTWU}(Y, l) \leq \text{WTWU}(X, l)
\end{equation}
\end{lemma}

\begin{proof}
$Y \supseteq X$であるから、$Y$を含むすべてのトランザクションは$X$も含む。したがって$\{T_t \mid Y \subseteq I(T_t)\} \subseteq \{T_t \mid X \subseteq I(T_t)\}$であり、部分集合上の$\text{TU}(T_t)$の総和は上位集合上の総和以下となる。
\end{proof}

\begin{theorem}[UDI枝刈り]
\label{thm:pruning}
$\max_{l} \text{WTWU}(X, l) < \theta_u$ならば、$X$は効用密集区間を持たず、$X$の任意の上位集合$Y \supseteq X$もまた効用密集区間を持たない。
\end{theorem}

\begin{proof}
性質~\ref{prop:twu-bound}より、すべての$l$に対して$\text{WU}(X, l) \leq \text{WTWU}(X, l) < \theta_u$であるから、$X$はすべての位置で効用条件に違反する。補題~\ref{lem:twu-antimonotone}より、すべての$l$に対して$\text{WTWU}(Y, l) \leq \text{WTWU}(X, l) < \theta_u$であるから、$Y$もまた枝刈りできる。
\end{proof}

\begin{lemma}[頻度に関する密集区間の反単調性]
\label{lem:freq-antimonotone}
アイテムセット$X$が頻度密集区間を持たないならば、$X$の上位集合$Y \supseteq X$も頻度密集区間を持たない。
\end{lemma}

\begin{proof}
$Y \supseteq X$より、$Y$を含むトランザクションは$X$を含むトランザクションの部分集合であるから、すべての$l$に対して$\sigma_W(Y, l) \leq \sigma_W(X, l)$が成り立つ。
\end{proof}

\begin{corollary}[同時枝刈り]
\label{cor:joint}
アイテムセット$X$は、(a) 頻度密集区間を持たない、または (b) すべてのウィンドウ位置にわたるWTWUの最大値が$\theta_u$未満である、のいずれかの場合に枝刈りできる。
\end{corollary}

\subsection{累積和による最適化}

各位置でウィンドウ効用をゼロから再計算することを避けるため、累積和を事前計算する。

\begin{definition}[効用累積和]
アイテムセット$X$に対して次を定義する：
\begin{equation}
    P_u(X, t) = \sum_{j=0}^{t-1} u(X, T_j)
\end{equation}
このとき$\text{WU}(X, l) = P_u(X, l{+}W{+}1) - P_u(X, l)$を$O(1)$時間で計算できる。
\end{definition}

同様に、TWU累積和により$O(1)$でウィンドウTWUクエリが可能となる：
\begin{equation}
    P_{\text{TWU}}(X, t) = \sum_{\substack{j=0 \\ X \subseteq I(T_j)}}^{t-1} \text{TU}(T_j)
\end{equation}

\subsection{アルゴリズム：UDI-Miner}

Algorithm~\ref{alg:udi-miner}に本レベルワイズマイニング手順を示す。

\begin{algorithm}[t]
\caption{UDI-Miner：効用密集区間マイニング}
\label{alg:udi-miner}
\begin{algorithmic}[1]
\REQUIRE データベース$\mathcal{D}$、外部効用$p$、ウィンドウサイズ$W$、頻度閾値$\theta_f$、効用閾値$\theta_u$、最大長$k_{\max}$
\ENSURE 効用密集区間ペア$(X, [s,e])$の集合

\STATE アイテム-トランザクション対応表$M$とトランザクション索引を構築
\STATE \textbf{// フェーズ1：単一アイテム}
\FOR{各アイテム$i \in \mathcal{I}$}
    \STATE $\text{TWU}(i)$をグローバルに計算
    \IF{$\text{TWU}(i) < \theta_u$}
        \STATE アイテム$i$を枝刈り \COMMENT{定理~\ref{thm:pruning}}
        \STATE \textbf{continue}
    \ENDIF
    \STATE スライディングウィンドウにより頻度密集区間$\mathcal{F}_i$を計算
    \IF{$\mathcal{F}_i = \emptyset$}
        \STATE \textbf{continue}
    \ENDIF
    \STATE 累積和$P_u(\{i\}, \cdot)$および$P_{\text{TWU}}(\{i\}, \cdot)$を構築
    \STATE $\mathcal{F}_i$内の効用密集部分区間を探索
    \IF{効用密集区間が見つかった場合}
        \STATE $\{i\}$を結果および$L_1$に追加
    \ENDIF
\ENDFOR

\STATE \textbf{// フェーズ2以降：複数アイテム候補}
\FOR{$k = 2$ \TO $k_{\max}$}
    \STATE $C_k \gets \textsc{AprioriGen}(L_{k-1})$
    \STATE $C_k \gets \textsc{Prune}(C_k, L_{k-1})$ \COMMENT{Apriori性質}
    \STATE $L_k \gets \emptyset$
    \FOR{各候補$X \in C_k$}
        \STATE 単一アイテム区間から許容範囲を計算
        \IF{許容範囲が空}
            \STATE \textbf{continue}
        \ENDIF
        \STATE $P_{\text{TWU}}(X, \cdot)$を構築
        \IF{$\max_l (\text{WTWU}(X, l)) < \theta_u$}
            \STATE \textbf{continue} \COMMENT{TWU枝刈り}
        \ENDIF
        \STATE 共起タイムスタンプを計算
        \STATE 許容範囲内で頻度密集区間を探索
        \STATE $P_u(X, \cdot)$を構築
        \STATE 効用密集部分区間を探索
        \IF{効用密集区間が見つかった場合}
            \STATE $X$を結果および$L_k$に追加
        \ENDIF
    \ENDFOR
    \IF{$L_k = \emptyset$}
        \STATE \textbf{break}
    \ENDIF
\ENDFOR
\end{algorithmic}
\end{algorithm}

\subsection{計算量解析}

\begin{theorem}[時間計算量]
$n = |\mathcal{D}|$、$m = |\mathcal{I}|$、$K$を評価された候補数とする。UDI-Minerの時間計算量は$O(K \cdot n + m \cdot n)$である。
\end{theorem}

\begin{proof}
アイテム-トランザクション対応表の構築には$O(n \cdot \bar{d})$を要する。ここで$\bar{d}$は平均トランザクションサイズである。各候補に対して、累積和の構築は$O(n)$、ウィンドウ走査は$O(n)$である。密集区間の計算は$O(c \log c)$で、$c$は出現回数である。レベル$k$での候補生成は$O(|L_{k-1}|^2)$で抑えられる。支配的なコストは、すべての候補にわたる累積和構築と効用検査の$O(K \cdot n)$である。
\end{proof}

累積和最適化により、各ウィンドウの効用クエリが$O(W)$から$O(1)$に削減され、素朴なウィンドウ走査と比較して$W$倍の高速化が達成される。
