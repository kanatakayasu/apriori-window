% ==============================================================
\section{問題定義}\label{sec:problem}
% ==============================================================

密集区間マイニングフレームワークを基盤とし，パターンの\emph{不在}および
時間レジーム間の\emph{構造変化}を研究するための対称的拡張を導入する．

\subsection{記法と準備}

$\mathcal{I} = \{i_1, i_2, \ldots, i_m\}$を有限のアイテム全体集合，
$D = [d_1, d_2, \ldots, d_N]$を時間順に並んだトランザクション列とする．
ここで各$d_t \subseteq \mathcal{I}$である．
アイテムセット（パターン）$P \subseteq \mathcal{I}$がトランザクション
$d_t$に\emph{出現}するとは，$P \subseteq d_t$が成り立つことをいう．

\begin{definition}[サポート時系列]\label{def:support_series}
パターン$P$，ウィンドウサイズ$W \in \mathbb{Z}_{>0}$，長さ$N$の
トランザクション列$D$に対し，\emph{サポート時系列}
$s_P = [s_P(0), s_P(1), \ldots, s_P(N-W)]$を以下で定義する：
\begin{equation}
s_P(t) = \bigl|\{d_j \in D \mid t \leq j < t + W,\; P \subseteq d_j\}\bigr|.
\label{eq:support_series}
\end{equation}
\end{definition}

\begin{definition}[密集区間~\cite{dong2012contrast}]\label{def:dense_interval}
閾値$\theta \in \mathbb{Z}_{>0}$が与えられたとき，パターン$P$の\emph{密集区間}とは，
すべての$t \in [s, e]$に対して$s_P(t) \geq \theta$が成り立つ極大連続範囲$[s, e]$である．
$P$のすべての密集区間の集合を$\mathcal{D}(P, W, \theta)$と表記する．
\end{definition}

\subsection{反密集区間}

\begin{definition}[反密集区間]\label{def:anti_dense}
\emph{低閾値}$\theta_{\text{low}} \in \mathbb{Z}_{>0}$
（通常$\theta_{\text{low}} < \theta$）が与えられたとき，
パターン$P$の\emph{反密集区間}とは，以下を満たす極大連続範囲$[s, e]$である：
\begin{equation}
s_P(t) < \theta_{\text{low}} \quad \forall\, t \in [s, e].
\label{eq:anti_dense}
\end{equation}
$P$のすべての反密集区間の集合を
$\overline{\mathcal{D}}(P, W, \theta_{\text{low}})$と表記する．
\end{definition}

反密集区間は密集区間の\emph{対称双対}である：
$\mathcal{D}$が持続的な高サポート期間を捉えるのに対し，
$\overline{\mathcal{D}}$は持続的な低サポート期間を捉える．
$\theta_{\text{low}} < \theta$の場合，補集合$[0, N-W] \setminus \mathcal{D}$は
$\overline{\mathcal{D}}$と\emph{同値ではない}ことに注意されたい．
これは$\theta_{\text{low}} \leq s_P(t) < \theta$となる中間帯域が
存在するためである．

\begin{theorem}[反密集区間の単調性]\label{thm:anti_monotone}
$P \subset Q$（すなわち$P$が$Q$の真部分集合）のとき，任意の反密集区間
$[s, e] \in \overline{\mathcal{D}}(P, W, \theta_{\text{low}})$に対して，
$[s, e] \subseteq [s', e']$を満たす反密集区間
$[s', e'] \in \overline{\mathcal{D}}(Q, W, \theta_{\text{low}})$が存在する．
すなわち，\textbf{部分集合の反密集区間は上位集合の反密集区間に包含される}．
\end{theorem}

\begin{proof}
$P \subset Q$であるから，$Q \subseteq d_t$ならば$P \subseteq d_t$が成り立つ．
したがって，任意のウィンドウ位置$t$に対して
\begin{equation}
s_P(t) \geq s_Q(t) \label{eq:support_monotone}
\end{equation}
が成り立つ（サポートの単調性）．

$[s, e] \in \overline{\mathcal{D}}(P, W, \theta_{\text{low}})$を$P$の任意の反密集区間とする．
定義より，すべての$t \in [s, e]$に対して$s_P(t) < \theta_{\text{low}}$が成り立つ．
式~(\ref{eq:support_monotone})より$s_Q(t) \leq s_P(t) < \theta_{\text{low}}$であるから，
すべての$t \in [s, e]$に対して$s_Q(t) < \theta_{\text{low}}$が成り立つ．
したがって$[s, e]$は$Q$のある極大反密集区間$[s', e']$に包含される
（$[s, e] \subseteq [s', e']$）．

すなわち，$P$の反密集区間は常に$Q \supset P$の反密集区間に包含されるため，
\emph{上位集合の反密集区間は少なくとも同じ幅を持つ}．
この性質は\emph{トップダウン}探索を可能にする：
上位集合$Q$が候補領域に反密集区間を持たなければ，
$Q$のいかなる部分集合$P$もそこに反密集区間を持ちえない
（対偶：$P$が反密集区間を持つならば$Q$も同領域に反密集区間を持つ）．
\end{proof}

\subsection{コントラスト密集パターン}

ここで二つの時間レジーム間で密集区間構造を比較するフレームワークを定義する．

\begin{definition}[時間レジーム]\label{def:regime}
\emph{時間レジーム}とはトランザクション列の連続部分列である．
\emph{レジーム境界}$\tau \in \{1, \ldots, N-1\}$が与えられたとき，
\emph{前レジーム}$R_1 = [1, \tau]$と\emph{後レジーム}$R_2 = [\tau+1, N]$を定義する．
\end{definition}

\begin{definition}[密集区間構造]\label{def:dis}
パターン$P$とレジーム$R_k$に対し，$R_k$に制限された\emph{密集区間構造}は
\begin{equation}
\mathcal{D}_k(P) = \{[s, e] \in \mathcal{D}(P, W, \theta) \mid [s, e] \cap R_k \neq \emptyset\},
\end{equation}
を$R_k$上に射影したもの（各区間を$R_k$にクリップしたもの）である．
レジーム$R_k$における\emph{密集カバレッジ}は
\begin{equation}
\text{cov}_k(P) = \frac{\sum_{[s,e] \in \mathcal{D}_k(P)} (e - s + 1)}{|R_k|}
\end{equation}
で定義する．
\end{definition}

\begin{definition}[パターンのトポロジー変化]\label{def:topology_change}
パターン$P$とレジーム境界$\tau$が与えられたとき，$P$の\emph{トポロジー変化タイプ}は
以下のように分類される：
\begin{enumerate}
\item \textbf{出現}：$\mathcal{D}_1(P) = \emptyset$かつ
  $\mathcal{D}_2(P) \neq \emptyset$．
  パターンが$R_1$では反密集（または不在）であり，$R_2$で密集となる．
\item \textbf{消失}：$\mathcal{D}_1(P) \neq \emptyset$かつ
  $\mathcal{D}_2(P) = \emptyset$．
  パターンが$R_1$では密集であり，$R_2$で反密集となる．
\item \textbf{増幅}：$\mathcal{D}_1(P) \neq \emptyset$，
  $\mathcal{D}_2(P) \neq \emptyset$，かつ
  マージン$\delta > 0$に対して$\text{cov}_2(P) > \text{cov}_1(P) + \delta$．
\item \textbf{収縮}：$\mathcal{D}_1(P) \neq \emptyset$，
  $\mathcal{D}_2(P) \neq \emptyset$，かつ
  $\text{cov}_1(P) > \text{cov}_2(P) + \delta$．
\item \textbf{安定}：上記のいずれにも該当しない．構造は本質的に不変．
\end{enumerate}
\end{definition}

\begin{remark}[マージン$\delta$の選択]
マージン$\delta$は増幅・収縮を安定と区別する感度パラメータである．
$\delta = 0$はカバレッジの任意の差異を構造変化と見なし（偽陽性増加），
$\delta$が大きすぎると真の変化を見逃す（偽陰性増加）．
実用的なガイドラインとして，
$\delta$を帰無仮説下（置換検定）のカバレッジ差の95\%点に設定する方法が
考えられるが，本研究では固定値を用いる．
$\delta$の自動選択は今後の課題である．
\end{remark}

\begin{definition}[コントラスト密集パターン]\label{def:cdp}
パターン$P$がレジーム境界$\tau$に関する\emph{コントラスト密集パターン}であるとは，
そのトポロジー変化タイプが安定\emph{ではない}ことをいう．
すべてのコントラスト密集パターンの集合を$\mathcal{C}(\tau, \delta)$と表記する．
\end{definition}

\subsection{構造変化の統計的有意性}

観測されたトポロジー変化がランダムな変動ではなく統計的に有意であるかを
評価するために，置換検定を用いる．

\begin{definition}[コントラスト統計量]\label{def:contrast_stat}
パターン$P$とレジーム境界$\tau$に対し，\emph{コントラスト統計量}を以下で定義する：
\begin{equation}
\Delta(P, \tau) = \text{cov}_2(P) - \text{cov}_1(P).
\label{eq:contrast_stat}
\end{equation}
$\Delta$が大きな正値のとき出現または増幅を，大きな負値のとき消失または収縮を示す．
\end{definition}

\begin{definition}[構造変化の置換検定]\label{def:permtest}
$\mathbf{t}_P = [t_1, t_2, \ldots, t_n]$をパターン$P$の出現タイムスタンプとする．
帰無仮説$H_0$：「レジーム境界$\tau$は$P$の密集区間構造に影響を与えない」の下で，
$\mathbf{t}_P$の$B$回のランダム置換（または$\tau$のランダムシフトと同値）を生成し，
置換コントラスト統計量$\Delta^{(1)}, \ldots, \Delta^{(B)}$を計算する．
置換$p$値は
\begin{equation}
p = \frac{1 + \sum_{b=1}^{B} \mathbf{1}[|\Delta^{(b)}| \geq |\Delta(P, \tau)|]}{1 + B}
\label{eq:perm_pvalue}
\end{equation}
で与えられる．
すべての検定対象パターンに対してBenjamini--Hochberg補正~\cite{benjamini1995fdr}を
適用した上で，$p \leq \alpha$のとき水準$\alpha$で$H_0$を棄却する．
\end{definition}

\begin{remark}[交換可能性の仮定]\label{rem:exchangeability}
上記の置換検定はタイムスタンプの交換可能性を暗黙に仮定する．
すなわち，帰無仮説のもとでパターンの出現位置が一様にランダムであることを
要求する．
実データではトレンドや季節性により出現確率が時間依存する場合があり，
この仮定が破られる可能性がある．
より頑健な代替案として，
(i) ブロック置換（時間的局所性を保持する置換），
(ii) 円形シフト検定（時系列の自己相関構造を保存する），
(iii) レジーム境界$\tau$のランダムシフト（タイムスタンプではなく
境界位置を置換する）が考えられる．
本研究では実装の単純さから全置換を採用するが，
実データ適用時には(ii)または(iii)への切替を推奨する．
\end{remark}

\subsection{計算量}

\begin{theorem}[反密集区間検出の計算量]\label{thm:complexity_anti}
パターン$P$の出現タイムスタンプ$\mathbf{t}_P$（$|\mathbf{t}_P| = n$）と
ウィンドウサイズ$W$が与えられたとき，反密集区間
$\overline{\mathcal{D}}(P, W, \theta_{\text{low}})$は
$O(n \log n)$時間で計算できる．
二ポインタ法を用いる場合は$O(n)$に改善される．
これは密集区間検出と同一の計算量クラスである．
\end{theorem}

\begin{proof}
$O(n)$個の臨界位置（タイムスタンプ$t_i$，$t_i - W$，$t_i + 1$，
および系列の両端）においてウィンドウサポートを評価する．
各評価は二分探索により$O(\log n)$時間で完了するため，
総計算量は$O(n \log n)$である．
二ポインタ法によりウィンドウカウントを増分的に更新する場合，
各タイムスタンプは高々2回（ウィンドウへの進入と退出）参照され，
$O(n)$に改善される．
密集区間検出からの変更点は，$\geq \theta$の代わりに
$< \theta_{\text{low}}$をチェックし，
閾値未達領域の開始・終了を追跡することのみである．
\end{proof}

\begin{theorem}[構造比較の完全性]\label{thm:completeness}
トポロジー変化分類体系（定義~\ref{def:topology_change}）は\emph{完全}である：
すべてのパターン$P$は五つのカテゴリ（出現・消失・増幅・収縮・安定）の
いずれか一つにのみ分類される．
\end{theorem}

\begin{proof}
五つの場合は二つのブール条件
（$\mathcal{D}_1 = \emptyset$，$\mathcal{D}_2 = \emptyset$）と
一つの比較（$\text{cov}_2 - \text{cov}_1$対$\pm\delta$）による
決定木で定義される．
\begin{itemize}
\item $\mathcal{D}_1 = \emptyset \land \mathcal{D}_2 = \emptyset$の場合：
  両方空$\Rightarrow$安定（いずれのレジームにも密集区間なし）．
\item $\mathcal{D}_1 = \emptyset \land \mathcal{D}_2 \neq \emptyset$の場合：
  出現．
\item $\mathcal{D}_1 \neq \emptyset \land \mathcal{D}_2 = \emptyset$の場合：
  消失．
\item $\mathcal{D}_1 \neq \emptyset \land \mathcal{D}_2 \neq \emptyset$の場合：
  \begin{itemize}
  \item $\text{cov}_2 > \text{cov}_1 + \delta$：増幅．
  \item $\text{cov}_1 > \text{cov}_2 + \delta$：収縮．
  \item その他：安定．
  \end{itemize}
\end{itemize}
これらの場合は相互排他的かつ網羅的である．\qed
\end{proof}
