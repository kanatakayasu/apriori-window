\section{二段階マイニングアルゴリズム}\label{sec:algorithm}
% ==============================================================

\S\ref{sec:problem}で定義した枠組みを実装する二段階稀少密集パターンマイニングアルゴリズムを提示する。
本アルゴリズムは二つの逐次的段階から成る：第1段階でスライディングウィンドウAprioriにより局所的に密集するすべてのパターンを発見し、第2段階で大域的に稀少なパターンのみを保持するようフィルタリングする。

\subsection{第1段階：密集区間発見}

第1段階はAprioriアルゴリズムを改変し、大域サポートの代わりに局所密集性を枝刈り基準として用いる。
定理~\ref{thm:weak_antimon}により確立された重要な知見は、局所密集性条件が反単調であること、すなわちアイテムセットが密集区間を持たなければその上位集合も持たないことである。

アルゴリズム~\ref{alg:phase1}にこの段階を記述する。
各候補アイテムセットに対して、共起タイムスタンプ（アイテムセット中のすべてのアイテムが同時に出現するトランザクションインデックス）を計算し、スライディングウィンドウ密集区間計算を適用する。

\begin{algorithm}[t]
\caption{第1段階：密集区間発見}
\label{alg:phase1}
\begin{algorithmic}[1]
\REQUIRE トランザクション列 $D$、ウィンドウサイズ $W$、密集閾値 $\theta$、最大アイテムセットサイズ $k_{\max}$
\ENSURE 局所的に密集するアイテムセットの集合 $\mathcal{F}_{\text{local}}$ とその密集区間

\STATE アイテム$\to$トランザクションIDマップ $M$ を構築
\STATE $\mathcal{F}_{\text{local}} \gets \emptyset$
\STATE $L_1 \gets \emptyset$ \COMMENT{密集1-アイテムセット}

\FORALL{アイテム $i \in \mathcal{I}$}
    \STATE $\tau_i \gets M[i]$ \COMMENT{ソート済みトランザクションID}
    \STATE $\mathcal{D}_i \gets \textsc{DenseIntervals}(\tau_i, W, \theta)$
    \IF{$\mathcal{D}_i \neq \emptyset$}
        \STATE $\mathcal{F}_{\text{local}}[\{i\}] \gets \mathcal{D}_i$
        \STATE $L_1 \gets L_1 \cup \{\{i\}\}$
    \ENDIF
\ENDFOR

\STATE $k \gets 2$
\WHILE{$L_{k-1} \neq \emptyset$ \AND $k \leq k_{\max}$}
    \STATE $C_k \gets \textsc{AprioriGen}(L_{k-1})$
    \STATE $C_k \gets \textsc{Prune}(C_k, L_{k-1})$
    \STATE $L_k \gets \emptyset$
    \FORALL{$P \in C_k$}
        \STATE $R \gets \bigcap_{i \in P} \mathcal{D}_i$ \COMMENT{単体アイテムからの候補範囲}
        \STATE $R \gets \{[s,e] \in R \mid e - s \geq W\}$
        \IF{$R = \emptyset$}
            \STATE \textbf{continue} \COMMENT{有効な候補範囲なし}
        \ENDIF
        \STATE $\tau_P \gets \bigcap_{i \in P} M[i]$ \COMMENT{共起タイムスタンプ}
        \STATE $\mathcal{D}_P \gets \textsc{DenseIntervals}(\tau_P, W, \theta, R)$
        \IF{$\mathcal{D}_P \neq \emptyset$}
            \STATE $\mathcal{F}_{\text{local}}[P] \gets \mathcal{D}_P$
            \STATE $L_k \gets L_k \cup \{P\}$
        \ENDIF
    \ENDFOR
    \STATE $k \gets k + 1$
\ENDWHILE

\RETURN $\mathcal{F}_{\text{local}}$
\end{algorithmic}
\end{algorithm}

\paragraph{候補範囲の最適化}
17行目では、候補中のすべての単体アイテムの密集区間を交差させる。
これにより探索を各構成アイテムが個別に密集している時間範囲に制限し、タイムライン全体にわたる高コストなスライディングウィンドウ計算を回避する。
この最適化は、いずれの構成アイテムもある領域で密集していなければ、複数アイテムのパターンもそこでは密集し得ないという反単調性により正当化される。

\paragraph{密集区間計算}
\textsc{DenseIntervals}サブルーチン（アルゴリズム~\ref{alg:dense_intervals}）は、ソート済み出現タイムスタンプ列に対するスライディングウィンドウ走査により、密集区間（定義~\ref{def:dense_interval}）を $O(n_P)$ で計算する。
候補範囲 $R$ が与えられた場合、走査はその範囲に制限され、さらに計算量を削減する。

\begin{algorithm}[t]
\caption{\textsc{DenseIntervals}: 密集区間計算}
\label{alg:dense_intervals}
\begin{algorithmic}[1]
\REQUIRE ソート済み出現タイムスタンプ列 $\tau = [\tau_1, \ldots, \tau_n]$、ウィンドウサイズ $W$、密集閾値 $\theta$
\ENSURE 密集区間の集合 $\mathcal{D}$
\STATE $\mathcal{D} \gets \emptyset$
\STATE $\text{left} \gets 0$, $\text{in\_dense} \gets \text{false}$, $s \gets -1$
\FOR{$l = 1$ \TO $N - W + 1$}
    \WHILE{$\text{left} < n$ \AND $\tau[\text{left}] < l$}
        \STATE $\text{left} \gets \text{left} + 1$
    \ENDWHILE
    \STATE $\text{right} \gets$ ウィンドウ $[l, l{+}W)$ 内の最後の要素位置
    \STATE $\text{count} \gets \text{right} - \text{left} + 1$
    \IF{$\text{count} \geq \theta$}
        \IF{$\neg\, \text{in\_dense}$}
            \STATE $s \gets l$, $\text{in\_dense} \gets \text{true}$
        \ENDIF
    \ELSE
        \IF{$\text{in\_dense}$}
            \STATE $\mathcal{D} \gets \mathcal{D} \cup \{[s, l{-}1]\}$, $\text{in\_dense} \gets \text{false}$
        \ENDIF
    \ENDIF
\ENDFOR
\IF{$\text{in\_dense}$}
    \STATE $\mathcal{D} \gets \mathcal{D} \cup \{[s, N{-}W{+}1]\}$
\ENDIF
\RETURN $\mathcal{D}$
\end{algorithmic}
\end{algorithm}

\subsection{第2段階：大域的稀少性フィルタリング}

第2段階は単純である：第1段階で発見された各パターンに対して大域サポートを計算し、最大サポート閾値 $\sigma_{\max}$ を超えるパターンを除外する。

\begin{algorithm}[t]
\caption{第2段階：大域的稀少性フィルタリング}
\label{alg:phase2}
\begin{algorithmic}[1]
\REQUIRE 局所的に密集するアイテムセット $\mathcal{F}_{\text{local}}$、トランザクション列 $D$、最大サポート $\sigma_{\max}$
\ENSURE 稀少密集パターン $\mathcal{R}$

\STATE $\mathcal{R} \gets \emptyset$
\FORALL{$(P, \mathcal{D}_P) \in \mathcal{F}_{\text{local}}$}
    \STATE $g \gets |\{t \mid P \subseteq d_t\}| / N$
    \IF{$g < \sigma_{\max}$}
        \STATE $\mathcal{R}[P] \gets \mathcal{D}_P$
    \ENDIF
\ENDFOR
\RETURN $\mathcal{R}$
\end{algorithmic}
\end{algorithm}

3行目の大域サポート計算はパターンあたり $O(N \cdot |P|)$ である。
$|\mathcal{F}_{\text{local}}|$ は通常小さい（局所的に密集するパターン数で制限される）ため、この段階は高速である。

\subsection{正当性と計算量}

\begin{corollary}[アルゴリズムの正当性]
アルゴリズム~\ref{alg:phase1}と~\ref{alg:phase2}を合わせて、定義~\ref{def:rdp}で定義された $\mathcal{R}(D, W, \theta, \sigma_{\max})$ を正確に計算する。
\end{corollary}

\begin{proof}
定理~\ref{thm:completeness}から直ちに従う。
第1段階は反単調な局所密集性基準（定理~\ref{thm:weak_antimon}）を用いたAprioriにより、すべての局所的に密集するパターンを発見する。
第2段階は大域的稀少性フィルタを適用する。
\end{proof}

\paragraph{時間計算量}
定理~\ref{thm:complexity}に述べた通り、全体の時間計算量は $O\bigl(\sum_{k=1}^{k_{\max}} \sum_{P \in C_k} n_P\bigr)$ であり、$n_P$ はパターン $P$ の出現回数である。
Apriori枝刈りにより $C_k$ は大幅に削減される：密集区間を持たないアイテムセットとそのすべての上位集合が除去される。

\paragraph{空間計算量}
アルゴリズムはアイテム-トランザクションIDマップ（$O(N \cdot \bar{w})$、$\bar{w}$ は平均トランザクション幅）、現在と前のレベルのアイテムセット（$O(C_k)$）、および出力（$O(|\mathcal{R}|)$）を格納する。
密集区間計算は $O(N)$ の一時的空間を使用する。

\subsection{パラメータ選択}

アルゴリズムは4つのパラメータを必要とする：ウィンドウサイズ $W$、密集閾値 $\theta$、最大サポート $\sigma_{\max}$、最大アイテムセットサイズ $k_{\max}$。

\begin{itemize}
\item \textbf{ウィンドウサイズ $W$}：時間解像度を制御する。小さい $W$ はより鋭いバーストを検出し、大きい $W$ はより広いトレンドを捕捉する。アプリケーションの時間粒度（例：小売業では日、ネットワークログでは秒）に基づいて設定すべきである。

\item \textbf{密集閾値 $\theta$}：ウィンドウ内の最小共起回数。$\theta / W$ は最小局所サポート率を表す。非自明なRDPが存在するためには $\theta / W > \sigma_{\max}$ を満たす必要がある。

\item \textbf{最大サポート $\sigma_{\max}$}：大域的稀少性フィルタを制御する。低い値ほどより稀少なパターンを生成する。典型的な範囲：0.01--0.10。

\item \textbf{最大アイテムセットサイズ $k_{\max}$}：探索の深さを制限する。$k_{\max} = 3$ または $4$ に設定すれば、多くのアプリケーションで十分であり計算も実行可能である。実用上、稀少密集パターンの主要なユースケースは $|P| = 2$（2アイテムセット）である。稀少なアイテムの共起は組合せ的に爆発し、$|P| \geq 3$ ではベースサポートがさらに低下するため密集区間の検出が困難になる。2アイテムセットの稀少密集パターンは、プロモーションバンドル（2商品セット）や協調的異常（2イベント共起）など、最も解釈しやすく実務的価値の高いパターンクラスに対応する。
\end{itemize}

RDPクラスが空でないためには、制約 $\theta / W > \sigma_{\max}$ が必要である。
直感的に、パターンが局所的に密集しかつ大域的に稀少であるためには、その出現が一様に分布するのではなく時間的に集中している必要がある。
