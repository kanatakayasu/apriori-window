\section{提案手法：DP-DIM}\label{sec:method}

本節では、差分プライバシー密集区間マイニング（DP-DIM）フレームワークを提示する。まず窓感度の厳密な解析を行い、次にDP密集区間検出アルゴリズムを提示し、閾値安定性・SVT変種・Apriori多レベル探索における予算合成を順に論じる。

\subsection{窓感度の解析}

密集区間検出にDPを適用する際の第一歩は、ウィンドウカウントクエリのグローバル感度を決定することである。感度が大きいほど必要なノイズ量が増加し、精度が低下するため、感度の正確な評価は極めて重要である。

\begin{definition}[窓感度（Window Sensitivity）]\label{def:window-sensitivity}
固定されたアイテムセット$X$と固定されたウィンドウ位置$l$に対するウィンドウカウント関数$f_{X,l}(\mathcal{D}) = C_W(X, l; \mathcal{D})$の$L_1$-グローバル感度を窓感度$\Delta_W$と定義する：
\[
\Delta_W = \max_{\mathcal{D} \sim \mathcal{D}'} |C_W(X, l; \mathcal{D}) - C_W(X, l; \mathcal{D}')|
\]
ここで$\mathcal{D} \sim \mathcal{D}'$は定義\ref{def:adjacency}の有界隣接関係である。
\end{definition}

\begin{theorem}[窓感度]\label{thm:window-sensitivity}
任意のアイテムセット$X$、ウィンドウサイズ$W$、およびウィンドウ位置$l$に対して、窓感度は$\Delta_W = 1$である。
\end{theorem}

\begin{proof}
隣接データベース$\mathcal{D} \sim \mathcal{D}'$を考える。定義\ref{def:adjacency}より、あるタイムスタンプ$t^*$においてのみ$T_{t^*} \neq T'_{t^*}$であり、他の全ての$t \neq t^*$について$T_t = T'_t$が成り立つ。

ウィンドウ位置$l$に対して、ウィンドウカウントの差は：
\[
|C_W(X, l; \mathcal{D}) - C_W(X, l; \mathcal{D}')| = |\mathbf{1}[X \subseteq T_{t^*}] - \mathbf{1}[X \subseteq T'_{t^*}]| \cdot \mathbf{1}[l \leq t^* \leq l + W]
\]
ここで$\mathbf{1}[\cdot]$は指示関数である。$t^*$がウィンドウ$[l, l+W]$に含まれない場合、差は0である。含まれる場合でも、指示関数の差は$\{-1, 0, 1\}$のいずれかの値を取る。最悪ケースは$X \subseteq T_{t^*}$かつ$X \not\subseteq T'_{t^*}$（またはその逆）のとき生じ、差は$\pm 1$となる。したがって$\Delta_W = 1$が成り立つ。

なお、各ウィンドウ位置のクエリ$f_{X,l}$を個別に見ると感度は1であるが、全ウィンドウ位置を連結したベクトルクエリ$\mathbf{f}_X(\mathcal{D}) = (C_W(X, 0), \ldots, C_W(X, n-W-1))$の$L_1$感度は、$t^*$が最大$\min(W+1, n-W)$個のウィンドウに寄与しうるため$\Delta_1 \mathbf{f}_X = \min(W+1, n-W)$となる。本アルゴリズムでは各ウィンドウ位置に独立にノイズを加えるため、個別クエリの感度$\Delta_W = 1$を用いる。
\end{proof}

\subsection{DP密集区間}

\begin{definition}[DP密集区間（DP Dense Interval）]\label{def:dp-dense}
ノイズ付きウィンドウカウント$\tilde{C}_W(X, l) = C_W(X, l) + \eta$（$\eta$はラプラスまたはガウスノイズ）に基づいて、$\tilde{C}_W(X, l) \geq \theta$を満たすウィンドウ左端$l$の極大連続区間をDP密集区間と定義する。
\end{definition}

\begin{algorithm}[t]
\caption{DP-DIM: 差分プライバシー密集区間検出}\label{alg:dp-dim}
\begin{algorithmic}[1]
\REQUIRE タイムスタンプ列$\mathbf{t}$、ウィンドウサイズ$W$、閾値$\theta$、プライバシパラメータ$\varepsilon$
\ENSURE DP密集区間集合$\mathcal{I}$
\STATE $L \leftarrow$ ユニークウィンドウ位置の集合
\STATE $\varepsilon_q \leftarrow \varepsilon / |L|$ \hfill\COMMENT{逐次合成による予算配分}
\STATE $b \leftarrow \Delta_W / \varepsilon_q$ \hfill\COMMENT{ラプラスノイズのスケール}
\STATE $\mathcal{I} \leftarrow \emptyset$
\STATE $\text{in\_dense} \leftarrow \text{false}$
\FOR{各ウィンドウ位置 $l \in L$}
  \STATE $c \leftarrow C_W(X, l)$ \hfill\COMMENT{真のカウント}
  \STATE $\tilde{c} \leftarrow c + \text{Lap}(b)$ \hfill\COMMENT{ノイズ付加}
  \IF{$\tilde{c} \geq \theta$}
    \IF{$\neg\text{in\_dense}$}
      \STATE $\text{start} \leftarrow l$
    \ENDIF
    \STATE $\text{end} \leftarrow l$
    \STATE $\text{in\_dense} \leftarrow \text{true}$
  \ELSE
    \IF{$\text{in\_dense}$}
      \STATE $\mathcal{I} \leftarrow \mathcal{I} \cup \{(\text{start}, \text{end})\}$
    \ENDIF
    \STATE $\text{in\_dense} \leftarrow \text{false}$
  \ENDIF
\ENDFOR
\IF{$\text{in\_dense}$}
  \STATE $\mathcal{I} \leftarrow \mathcal{I} \cup \{(\text{start}, \text{end})\}$
\ENDIF
\RETURN $\mathcal{I}$
\end{algorithmic}
\end{algorithm}

\begin{theorem}[DP-DIMのプライバシー保証]\label{thm:dp-dim-privacy}
アルゴリズム\ref{alg:dp-dim}は$\varepsilon$-差分プライバシーを満たす。
\end{theorem}

\begin{proof}
各ウィンドウ位置$l$でのクエリにラプラスメカニズム（$\varepsilon_q = \varepsilon/|L|$, 感度$\Delta_W = 1$）を適用する。逐次合成定理より、$|L|$個のクエリの合成は$|L| \cdot \varepsilon_q = \varepsilon$-DPを満たす。
\end{proof}

\subsection{閾値安定性}

ノイズ付加により、真のカウントが閾値近傍にある場合、密集/非密集の判定が不安定になる。この問題を制御するために閾値安定性の概念を導入する。

\begin{definition}[閾値安定性（Threshold Stability）]\label{def:stability}
ノイズ付きカウント$\tilde{c}$と閾値$\theta$に対して、安定性マージン$\gamma$を以下で定義する：
\[
\gamma = \frac{\Delta_W}{\varepsilon_q} \cdot \ln\frac{1}{1 - p}
\]
ここで$p$は所望の信頼度である。$|\tilde{c} - \theta| \geq \gamma$のとき、閾値判定は確率$p$以上で正しい。
\end{definition}

\begin{theorem}[閾値安定性保証]\label{thm:stability}
真のカウント$c$が$|c - \theta| \geq \gamma$を満たすとき、ラプラスメカニズムによるノイズ付きカウント$\tilde{c}$に基づく密集判定は確率$p$以上で正しい。
\end{theorem}

\begin{proof}
ラプラス分布$\text{Lap}(b)$（$b = \Delta_W / \varepsilon_q$）に対して、$\Pr[|\eta| < b \cdot \ln(1/(1-p))] = p$が成り立つ。
$|c - \theta| \geq \gamma = b \cdot \ln(1/(1-p))$のとき、$|\eta| < \gamma$ならば$\text{sign}(\tilde{c} - \theta) = \text{sign}(c - \theta)$であり、この事象の確率は$p$以上である。
\end{proof}

\subsection{Sparse Vector Technique}

ウィンドウ位置の多くが非密集であるスパースなケースでは、Sparse Vector Technique (SVT)~\cite{dwork2009complexity,lyu2017understanding}によりプライバシ予算を節約できる。SVTの核心的な洞察は、below-threshold の回答にはプライバシ予算を消費しないという点にある。

\begin{algorithm}[t]
\caption{SVT-DIM: SVTによるDP密集区間検出}\label{alg:svt-dim}
\begin{algorithmic}[1]
\REQUIRE ウィンドウカウント列、閾値$\theta$、予算$\varepsilon$、上限$c_\text{max}$
\ENSURE above-threshold のウィンドウ位置集合
\STATE $\varepsilon_1 \leftarrow \varepsilon / 2$, $\varepsilon_2 \leftarrow \varepsilon / 2$
\STATE $\tilde{\theta} \leftarrow \theta + \text{Lap}(\Delta_W / \varepsilon_1)$ \hfill\COMMENT{ノイズ閾値}
\STATE $\text{count} \leftarrow 0$
\FOR{各ウィンドウ位置 $l$}
  \STATE $\nu_l \leftarrow \text{Lap}(2c_\text{max} \cdot \Delta_W / \varepsilon_2)$ \hfill\COMMENT{クエリノイズ}
  \IF{$C_W(X, l) + \nu_l \geq \tilde{\theta}$}
    \STATE 出力: $l$は above-threshold
    \STATE $\text{count} \leftarrow \text{count} + 1$
    \IF{$\text{count} \geq c_\text{max}$}
      \STATE \textbf{停止}
    \ENDIF
  \ELSE
    \STATE 出力: $l$は below-threshold（$\bot$）
  \ENDIF
\ENDFOR
\end{algorithmic}
\end{algorithm}

SVTの総プライバシ消費は$\varepsilon$であり、above-threshold回数$c_\text{max}$に依存するがクエリ総数$|L|$には依存しない~\cite{lyu2017understanding}。密集区間が疎（$c_\text{max} \ll |L|$）である場合、SVTはラプラスメカニズムの逐次合成と比較してノイズスケールを大幅に削減できる。ただし、SVTは各クエリに対して above/below の二値判定のみを返すため、ノイズ付きカウント値そのものは取得できないという制約がある。

\subsection{滑らかな感度に基づく分析}\label{subsec:smooth-sensitivity}

グローバル感度は最悪ケースに基づくためノイズが過大となる場合がある。密集区間マイニングでは、密集区間内部のウィンドウカウントは閾値$\theta$を大きく上回る傾向があり、データインスタンスごとに必要なノイズ量を削減できる可能性がある。Nissim ら~\cite{nissim2007smooth}の\emph{滑らかな感度}（smooth sensitivity）の枠組みを密集区間検出に適用し、その改善余地を分析する。

\begin{definition}[ウィンドウカウントの局所感度]\label{def:local-sensitivity}
データベース$\mathcal{D}$に対するウィンドウカウントクエリ$f_{X,l}$の局所感度を以下で定義する：
\[
LS_{f_{X,l}}(\mathcal{D}) = \max_{\mathcal{D}' \sim \mathcal{D}} |f_{X,l}(\mathcal{D}) - f_{X,l}(\mathcal{D}')|
\]
ウィンドウカウントの場合、$LS_{f_{X,l}}(\mathcal{D}) \in \{0, 1\}$である。タイムスタンプ$t^*$がウィンドウ$[l, l+W]$に含まれかつ$X \subseteq T_{t^*}$であるような$t^*$が存在するとき局所感度は1、そうでないとき0となる。
\end{definition}

\begin{definition}[密集区間に対する$\beta$-滑らかな感度]\label{def:smooth-sensitivity-dim}
$\beta > 0$に対して、$\beta$-滑らかな感度を以下で定義する：
\[
S^*_{f_{X,l},\beta}(\mathcal{D}) = \max_{k \geq 0} e^{-k\beta} \cdot \max_{\mathcal{D}'': d(\mathcal{D}, \mathcal{D}'') = k} LS_{f_{X,l}}(\mathcal{D}'')
\]
ウィンドウカウントでは$LS \in \{0, 1\}$であるため、距離$k$のデータベースでウィンドウに影響する隣接データベースが存在するかの二値問題に帰着する。
\end{definition}

\begin{property}[密集ウィンドウの滑らかな感度の特徴]\label{prop:smooth-char}
ウィンドウ$[l, l+W]$内に$X$を含むトランザクションが$m$個存在するとき（$C_W(X,l) = m$）、$m \geq 1$ならば局所感度は1であり$S^*_{f_{X,l},\beta}(\mathcal{D}) = 1$となる。$m = 0$の場合、距離1のデータベースで$X$を含むトランザクションを注入できるため$S^*_{f_{X,l},\beta}(\mathcal{D}) \leq e^{-\beta}$となる。

密集ウィンドウ（$C_W(X,l) \geq \theta \geq 1$）に対しては常に$S^* = 1 = \Delta_W$であるため、滑らかな感度はグローバル感度と一致する。すなわち、\textbf{ウィンドウカウントの感度が既に$\Delta_W = 1$と小さいため、滑らかな感度による改善余地は限定的}であり、これは本問題の構造的な強みである。
\end{property}

\subsection{情報理論的下界}\label{subsec:lower-bound}

DP-DIMの精度限界を理論的に評価するため、差分プライバシー制約下での密集区間検出の情報理論的下界を分析する。

\begin{theorem}[DP密集区間検出の精度下界]\label{thm:lower-bound}
$n$タイムスタンプ、ウィンドウサイズ$W$、閾値$\theta$の密集区間検出問題において、$\varepsilon$-差分プライバシーを満たす任意のメカニズム$\mathcal{M}$に対して、単一ウィンドウ位置$l$における密集判定の誤り確率$p_{\text{err}}$は以下の下界を満たす：
\[
p_{\text{err}} \geq \frac{1}{2}\exp\left(-2\varepsilon \cdot |c - \theta|\right)
\]
ここで$c = C_W(X, l)$は真のウィンドウカウントである。特に$|c - \theta| \leq \frac{\ln 2}{2\varepsilon}$のとき$p_{\text{err}} \geq 1/4$となる。
\end{theorem}

\begin{proof}
バイナリ仮説検定の枠組みで考える。隣接データベースの連鎖によりカウントを1ずつ変化させることで、$C_W(X,l) = \theta + \delta_c$と$C_W(X,l) = \theta - \delta_c$（$\delta_c = |c - \theta|$）の2つのデータベースを距離$2\delta_c$で構成できる。差分プライバシーのグループプライバシー性質より、出力分布間の全変動距離は$1 - e^{-2\varepsilon\delta_c}$で上界される。Le Cam の補題~\cite{lecam1973convergence}により、最適検定の誤り確率は$p_{\text{err}} \geq \frac{1}{2}e^{-2\varepsilon\delta_c}$で下界される。
\end{proof}

\begin{corollary}[ラプラスメカニズムの準最適性]\label{cor:near-optimal}
ラプラスメカニズム（感度$\Delta_W = 1$）の閾値判定誤り確率は$\frac{1}{2}e^{-\varepsilon|c - \theta|}$であり、定理\ref{thm:lower-bound}の下界$\frac{1}{2}e^{-2\varepsilon|c-\theta|}$と比較すると、指数部で定数倍（2倍）の差しかない。すなわち、ラプラスメカニズムはウィンドウカウントの密集判定において\textbf{定数倍の意味で準最適}（near-optimal）である。
\end{corollary}

この下界解析は2つの重要な示唆を持つ。第一に、$\varepsilon$を小さくしても精度が急激に低下しない領域（$|c - \theta|$が大きい密集ウィンドウ）が存在し、E1実験（\S\ref{sec:experiments}）で観察される$\varepsilon$に対する精度の緩やかな変化と整合する。第二に、感度$\Delta_W = 1$という密集区間マイニングの構造的優位性により、ラプラスメカニズムが既に下界に近い性能を達成しており、更なるメカニズム改善の余地は本質的に小さいことを示している。

\subsection{プライバシ予算合成}\label{subsec:composition}

Apriori型の多レベル探索では、複数のアイテムセット候補に対して多数のウィンドウカウントクエリを発行する。この多段階クエリにおけるプライバシ予算の管理が、DP-DIMの実用性を左右する重要な要素である。

\begin{definition}[階層的予算配分]\label{def:composition}
Apriori型の$K$レベル探索において、総予算$\varepsilon$を以下の二段階で配分する：
\begin{enumerate}
  \item \textbf{レベル間配分}：レベル$k$（$k = 1, \ldots, K$）に$\varepsilon_k = \varepsilon / K$を配分する。
  \item \textbf{レベル内配分}：レベル$k$内の候補数を$m_k$とし、各候補に$\varepsilon_k / m_k$を配分する。
\end{enumerate}
\end{definition}

\begin{theorem}[多レベル探索のプライバシー保証]\label{thm:multi-level}
定義\ref{def:composition}の階層的予算配分による多レベルDP-DIMは、$\varepsilon$-差分プライバシーを満たす。
\end{theorem}

\begin{proof}
各候補アイテムセットについて、そのウィンドウカウントクエリ群はアルゴリズム\ref{alg:dp-dim}により$\varepsilon_k / m_k$-DPを満たす。レベル$k$内の$m_k$候補に対して逐次合成定理を適用すると、レベル$k$全体で$m_k \cdot (\varepsilon_k / m_k) = \varepsilon_k = \varepsilon / K$-DPを満たす。さらに$K$レベル間の逐次合成により、全体で$K \cdot (\varepsilon / K) = \varepsilon$-DPが保証される。
\end{proof}

高度合成定理（性質\ref{prop:advanced}）を適用すると、より厳密な上界が得られる。

\begin{corollary}[高度合成による予算効率]\label{cor:multi-level}
総クエリ数$M = \sum_{k=1}^{K} m_k \cdot |L_k|$（$|L_k|$はレベル$k$のウィンドウ位置数）に対して均一配分$\varepsilon_0 = \varepsilon / M$（逐次合成）とする。同じ$\varepsilon_0$で高度合成定理を適用すると、総プライバシ消費は：
\[
\varepsilon'_\text{adv} = \varepsilon_0 \sqrt{2M\ln(1/\delta')} + M\varepsilon_0(e^{\varepsilon_0} - 1)
\]
$\varepsilon_0 \ll 1$のとき$\varepsilon'_\text{adv} \approx \varepsilon_0\sqrt{2M\ln(1/\delta')} = O(\sqrt{M})$であり、逐次合成の$\varepsilon_\text{seq} = M\varepsilon_0 = O(M)$と比較して$O(\sqrt{M})$の改善となる。
\end{corollary}
